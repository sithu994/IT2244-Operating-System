Q7).Create simple calculator using "expr" command same as the Q3

[root@DESKTOP-73GSIST ~]# vi calculator.sh
[root@DESKTOP-73GSIST ~]# more calculator.sh
echo "Enter first number:"
read num1

echo "Enter second number:"
read num2

echo "Choose operation (+, -, \*, /, %):"
read op

case $op in
  +)
    result=$(expr $num1 + $num2)
    echo "Result: $result"
    ;;
  -)
    result=$(expr $num1 - $num2)
    echo "Result: $result"
    ;;
  \*)
    result=$(expr $num1 \* $num2)
    echo "Result: $result"
    ;;
  /)
    result=$(expr $num1 / $num2)
    echo "Result: $result"
    ;;
  %)
    result=$(expr $num1 % $num2)
    echo "Result: $result"
    ;;
  *)
    echo "Invalid operation!"
    ;;
esac

[root@DESKTOP-73GSIST ~]# chmod 777 calculator.sh
[root@DESKTOP-73GSIST ~]# ./calculator.sh
Enter first number:
4
Enter second number:
2
Choose operation (+, -, \*, /, %):
+
Result: 6
[root@DESKTOP-73GSIST ~]# ./calculator.sh
Enter first number:
4
Enter second number:
2
Choose operation (+, -, \*, /, %):
*
Result: 8
